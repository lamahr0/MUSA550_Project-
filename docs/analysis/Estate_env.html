def calculate_mean_sales(df,zip_codes):
    df = df[df[' ZIP CODE'].isin(zip_codes)]
    mean_sales_by_year = df.groupby('year')['SALE_PRICE_M'].mean().reset_index()
    return mean_sales_by_year
